## Bullet -- cambios en src/entities/bullet.py

Nueva firma de __init__:

`python
def __init__(
    self, x, y,
    launch_frames=None, travel_path=None, impact_frames=None,
    w=BULLET_W_L1, h=BULLET_H_L1,
    angle_deg: float = 0.0,           # NUEVO
    damage: int = BULLET_DAMAGE_DEFAULT,  # NUEVO
    tint: tuple | None = None,         # NUEVO
    speed_mult: float = 1.0,           # NUEVO para plasma
):
`

Movimiento con angulo:

`python
import math
rad = math.radians(angle_deg)
self._vx = math.sin(rad) * BULLET_SPEED * speed_mult
self._vy = -math.cos(rad) * BULLET_SPEED * speed_mult
self.damage = damage
`

update usa vx/vy:

`python
self.x += self._vx * dt
self.y += self._vy * dt
self.rect.x = int(self.x)
self.rect.y = int(self.y)
if self.rect.bottom < 0 or self.rect.right < 0 or self.rect.left > SCREEN_W:
    self.kill()
`

Tinting helper:

`python
def _apply_tint(surface, tint):
    tinted = surface.copy()
    tinted.fill(tint, special_flags=pygame.BLEND_RGBA_MULT)
    return tinted
`

Aplicar a todos los frames en __init__ si tint no es None.

## Testing

`python
def test_bullet_default_moves_straight_up():
    b = Bullet(270, 500)
    b.update(1.0)
    assert b.rect.y < 500

def test_bullet_angle_moves_diagonally():
    b = Bullet(270, 500, angle_deg=30.0)
    b.update(1.0)
    assert b.rect.x > 270 - b.w // 2

def test_bullet_damage_default():
    b = Bullet(270, 500)
    assert b.damage == BULLET_DAMAGE_DEFAULT

def test_bullet_custom_damage():
    b = Bullet(270, 500, damage=5)
    assert b.damage == 5
`
