## `AudioSystem` — `src/systems/audio.py`

Módulo singleton. Carga lazy con caché. Tolerante a archivos faltantes — si no existen, continúa sin crash.

```python
import pygame
from pathlib import Path
from src.settings.audio import MUSIC_VOLUME, SFX_VOLUME

_BASE = Path(__file__).parent.parent.parent / "assets"
_cache: dict[str, pygame.mixer.Sound] = {}
_muted: bool = False


def init() -> None:
    if pygame.mixer.get_init():
        return
    try:
        pygame.mixer.init(frequency=44100, size=-16, channels=2, buffer=512)
    except pygame.error:
        pass


def play_sfx(path: str) -> None:
    if _muted or not pygame.mixer.get_init():
        return
    if path not in _cache:
        full = _BASE / path
        if not full.exists():
            return
        try:
            _cache[path] = pygame.mixer.Sound(str(full))
        except pygame.error:
            return
    _cache[path].set_volume(SFX_VOLUME)
    _cache[path].play()


def play_music(path: str) -> None:
    if not pygame.mixer.get_init():
        return
    full = _BASE / path
    if not full.exists():
        return
    try:
        pygame.mixer.music.load(str(full))
        pygame.mixer.music.set_volume(MUSIC_VOLUME)
        pygame.mixer.music.play(-1)
    except pygame.error:
        pass


def stop_music() -> None:
    if pygame.mixer.get_init():
        pygame.mixer.music.stop()


def toggle_mute() -> None:
    global _muted
    _muted = not _muted
    if pygame.mixer.get_init():
        pygame.mixer.music.set_volume(0.0 if _muted else MUSIC_VOLUME)


def is_muted() -> bool:
    return _muted


def clear_cache() -> None:
    _cache.clear()
```

### `__init__`
```python
self._game_over_played = False

```

### `_reset()`
```python

```

### `process_input(events)` — agregar en el loop de eventos
```python
if event.type == pygame.KEYDOWN and event.key == pygame.K_m:
    audio.toggle_mute()
```

### `update(dt)` — puntos de llamada

| Evento | Llamada |
|--------|---------|
| Player dispara bala | `audio.play_sfx(SFX_SHOOT)` |
| Bala impacta enemigo | `audio.play_sfx(SFX_IMPACT)` |
| Enemigo muere | `audio.play_sfx(SFX_EXPLOSION)` |
| Boss muere | `audio.play_sfx(SFX_EXPLOSION)` |
| Boss recibe daño (sobrevive) | `audio.play_sfx(SFX_BOSS_HIT)` |
| Player recoge power-up | `audio.play_sfx(SFX_POWERUP)` |
| Player pierde vida | `audio.play_sfx(SFX_PLAYER_HIT)` |
| Game over (una vez) | `audio.play_sfx(SFX_GAME_OVER)` |

Game over se detecta así en `update()`:
```python
if self.player.lives <= 0:
    self.state = "game_over"
    if not self._game_over_played:
        audio.play_sfx(SFX_GAME_OVER)
        self._game_over_played = True
```

## Testing — `tests/test_audio.py`

```python
import src.systems.audio as audio


def test_starts_unmuted():
    assert not audio.is_muted()


def test_toggle_mute():
    audio.toggle_mute()
    assert audio.is_muted()
    audio.toggle_mute()
    assert not audio.is_muted()


def test_play_sfx_missing_file_no_crash():
    audio.play_sfx("audio/sfx/__nonexistent__.wav")


def test_play_sfx_muted_skips_cache():
    audio.clear_cache()
    audio.toggle_mute()
    audio.play_sfx("audio/sfx/__nonexistent__.wav")
    assert "audio/sfx/__nonexistent__.wav" not in audio._cache
    audio.toggle_mute()


def test_clear_cache_empties():
    audio.clear_cache()
    assert audio._cache == {}
```
